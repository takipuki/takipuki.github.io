## 03

1. rewrite with do-while

   ```c:line-numbers
   #include <stdio.h>

   int main() {
       int weeks = 2, days_in_week = 7;

       int i = 1;
       do {
           printf("Week: %d\n", i);

           int j = 1;
           do {
               if (i % 2 == 0) {
                   if (j % 2 == 0)
                       printf("Day: %d\n", j);
               }
               else{
                   if (j % 2 != 0)
                       printf("Day: %d\n", j);
               }
               j++;
           } while (j <= days_in_week);
           i++;
       } while (i <= weeks);

       return 0;
   }
   ```

   ```c:line-numbers
   #include <stdio.h>

   int main() {
       float sum_cg3 = 0;
       int count_cg3 = 0;

       float max = 0;
       float min = 5;
       int count_max = 0;

       float tmp;
       for (int i = 0; i < 100; i++) {
           scanf("%d", &tmp);

           if (tmp > 3) {
               sum_cg3 += tmp;
               count_cg3++;
           }
        
           if (tmp > max) {
               max = tmp;
               count_max = 1;
           } else if (tmp == max)
               count_max++;

           if (tmp < min) min = tmp;
       }

       float avg_cg3 = sum_cg3 / count_cg3;

       printf("Average: %.2f\n", avg_cg3);
       printf("Higest: %.2f and count: %d\n", max, count_max);
       printf("Lowest: %.2f\n", min);
   }
   ```
