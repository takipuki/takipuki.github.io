 1: #include <stdio.h>
 2: 
 3: int main() {
 4:     int x, y; scanf("%d%d", &x, &y);
 5: 
 6: /*
 7:  * the gcd is the largest number <= the smaller of the two
 8:  * that can devide both x and y;
 9:  * so we can start from the smaller one of the two and
10:  * keep decrementing that value until it can devide the two
11:  * e.g.
12:  * x = 12 and y = 9
13:  * take the smaller one y and begin tesing
14:  * 9 cannot devide the two; decrement 9
15:  * 8 cannot devide the two; decrement 8
16:  * 7 cannot devide the two; decrement 7
17:  * 6 cannot devide the two; decrement 6
18:  * 5 cannot devide the two; decrement 5
19:  * 4 cannot devide the two; decrement 4
20:  * 3 can devide the two
21:  * so 3 is the gcd
22:  */
23: 
24:     // taking the smaller one of the two
25:     // as the larger one cannot be the gcd
26:     int gcd = (x < y) ? x : y;
27:     for (; !(x % gcd == 0 && y % gcd == 0); gcd--);
28:     //       ^^^^^^^^^^^^^^^^^^^^^^^^^^^^   ^^^^^
29:     // if gcd cant devide the two^          ^decrement gcd
30: 
31:     printf("GCD: %d\n", gcd);
32: 
33: 
34: /*
35:  * the lcm of x and y has to be the multiple of both x and y;
36:  * so we can just test the multiples of any one of them (x or y);
37:  * just check whether the multiples of any one (x or y) is divisible by
38:  * the other number;
39:  * e.g.
40:  * x = 6 and y = 8
41:  * testing multiples of y
42:  * y * 1 = 8; 8 not divisible by x
43:  * go to next multiple of y
44:  * y * 2 = 16; 16 not divisible by x
45:  * go to next multiple of y
46:  * y * 3 = 24; 24 is divisible by x
47:  * so 24 is the lcm
48:  */
49: 
50:     // taking the larger one of the two
51:     // as the smaller one cannot be the lcm
52:     int max = (x > y) ? x : y; // max
53:     int lcm = max;
54:     for (; !(lcm % x == 0 && lcm % y == 0); lcm += max);
55:     //       ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^ ^^^^^^^^^^
56:     // if the two nums cant devide the lcm^ ^go to next multiple
57:     printf("LCM: %d\n", lcm);
58: }
