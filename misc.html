Convert decimal to base

 1: #include <stdio.h>
 2: 
 3: char symbols[36]; // maximum base 36; digits (0-9) and alpha (A-Z)
 4: 
 5: void dec_to_base(int n, int base) {
 6:     if (!n) return;
 7: 
 8:     dec_to_base(n/base, base);
 9:     putchar(symbols[n % base]);    // least significant bit so printing last
10: }
11: 
12: int main(void) {
13:     int base, n;
14:     scanf("%d %d", &base, &n);
15: 
16:     if (base < 2 || base > 36) {
17:         printf("Only 2 to 36 bases are supported.\n");
18:         return 0;
19:     }
20: 
21:     if (!n) {
22:         printf("0\n");
23:         return 0;
24:     }
25: 
26:     // initializing symbols
27:     for (int i = 0; i < base; i++)
28:         symbols[i] = (i >= 10) ? 'A' + i-10 : '0' + i;
29: 
30:     dec_to_base(n, base);
31:     putchar('\n');
32: }

Armstrong number

 1: #include <stdio.h>
 2: #include <math.h>
 3: 
 4: int main() {
 5:     int n; scanf("%d", &n);
 6: 
 7:     int n_digs = log10(n) + 1;
 8: 
 9:     int sum = 0;
10:     for (int i = n; i; i /= 10)
11:         sum += pow(i % 10, n_digs);
12: 
13:     if (sum == n)
14:         printf("Yes\n");
15:     else
16:         printf("No\n");
17: }
