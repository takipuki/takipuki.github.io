    if (determinant < 0) {
        printf("Imaginary\n");
        return 0;

#include <stdio.h>

int main(void) {
    int correct, guess;
    scanf("%d", &correct);

    for (int i = 1; i <= 3; i++) {
        scanf("%d", &guess);
        if (guess == correct) {
            printf("Right, Player-2 wins!\n");
            return 0;
        }
        printf("Wrong, %d chance(s) left!\n", 3-i);
    }

    printf("Player-1 wins!\n");
}
