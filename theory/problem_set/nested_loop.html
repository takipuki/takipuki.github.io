 5: 
 6:     // top half + middle line
 7:     int loop = n/2+1;
 8:     for (int i = 1; i <= loop; i++) {
 9:         for (int j = 0; j < loop-i; j++) putchar(' ');
10:         for (int j = 0; j < i     ; j++) putchar('*');
11:         for (int j = 0; j < i-1   ; j++) putchar('*');
12:         putchar('\n');
13:     }
14: 
15:     // bottom half; thus starting from loop-1
16:     for (int i = loop-1; i >= 1; i--) {
17:         for (int j = 0; j < loop-i; j++) putchar(' ');
18:         for (int j = 0; j < i     ; j++) putchar('*');
19:         for (int j = 0; j < i-1   ; j++) putchar('*');
