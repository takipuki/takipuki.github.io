int main() {
    int n; scanf("%d", &n);

    int max = 0,
        min = 99999,
        max_indx = 0,
        min_indx = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > max) {
            max = arr[i];
            max_indx = i;
        }
        if (arr[i] < min){
             min = arr[i];
             min_indx = i;
        }
    }

    printf("Max: %d, Index: %d\n", max, max_indx);
    printf("Min: %d, Index: %d\n", min, min_indx);
}

int main() {
    int n; scanf("%d", &n);
    int arr[n];
    for (int i = 0; i < n; i++) scanf("%d", arr+i);
    int query; scanf("%d", &query);

    int found_n = 0;
    int found_indc[n]; // max we need is n; all same
    for (int i = 0; i < n; i++)
        if (arr[i] == query)
            found_indc[found_n++] = i;

    if (found_n) {
        printf("FOUND at index position: ");
        for (int i = 0; i < found_n; i++)
            printf("%d ", found_indc[i]);
        putchar('\n');
    } else
        printf("NOT FOUND\n");
}

c
#include <stdio.h>

int main() {
    int n; scanf("%d", &n);
    int arr[n];
    for (int i = 0; i < n; i++) scanf("%d", arr+i);   

    int sorted = 0; // flag
    for (int i = 0; !sorted && i < n; i++) {
        sorted = 1;
        for (int j = 1; j < n-i; j++)
            if (arr[j-1] > arr[j]) {
                int tmp  = arr[j];
                arr[j]   = arr[j-1];
                arr[j-1] = tmp;
                sorted = 0; // had to swap so not sorted yet
            }
    }

    for (int i = 0; i < n; i++)
        printf("%d ", arr[i]);
    putchar('\n');
}

c
#include <stdio.h>

int main() {
    int n; scanf("%d", &n);
    int arr[n];
    for (int i = 0; i < n; i++) scanf("%d", arr+i);
    
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            if (arr[i] == arr[j]) {
                // matched; no need to remove, just shift left
                // [1 1 2 3] -> [1 2 3]
                // [1 1 2 3] -> [1 2 2 3] -> [1 2 3]
                //  k---^          k---^
                for (int k = j+1; k < n; k++)
                    arr[k-1] = arr[k];

                n--; // one gone
                j--; // [1 1 2 3] -> [1 2 3]
                     // j--^         j--^
                     // j will go to 3 on the next loop
                     // so decrement j so it starts from 2 again
            }
        }
    }

    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    putchar('\n');
}

#include <stdio.h>

int main() {
    int a_n; scanf("%d", &a_n);
    int a[a_n];
    for (int i = 0; i < a_n; i++) scanf("%d", a+i);

    int b_n; scanf("%d", &b_n);
    int b[b_n];
    for (int i = 0; i < b_n; i++) scanf("%d", b+i);

    int uni[a_n + b_n];
    int uni_n = 0;

    for (int i = 0; i < a_n; i++)
        uni[i] = a[i];
    uni_n = a_n;

    for (int i = 0; i < b_n; i++) {
        int found = 0;
        for (int j = 0; j < a_n; j++) {
            if (b[i] == uni[j]) {
                found = 1;
                break; // don't add
            }
        }
        if (!found) uni[uni_n++] = b[i];
    }

    if (!uni_n)
        printf("Empty set");
    else
        for (int i = 0; i < uni_n; i++) printf("%d ", uni[i]);
    putchar('\n');
}
