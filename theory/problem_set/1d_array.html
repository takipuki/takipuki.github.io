 1: #include <stdio.h>
 2: 
 3: int main() {
 4:     int n; scanf("%d", &n);
 5:     int arr[n];
 6:     for (int i = 0; i < n; i++) scanf("%d", arr+i);
 7: 
 8:     for (int i = 0; i < n; i++) {
 9:         for (int j = i+1; j < n; j++) {
10:             if (arr[i] == arr[j]) {
11:                 // matched; no need to remove, just shift left
12:                 // [1 1 2 3] -> [1 2 3]
13:                 // [1 1 2 3] -> [1 2 2 3] -> [1 2 3]
14:                 //  k---^          k---^
15:                 for (int k = j+1; k < n; k++)
16:                     arr[k-1] = arr[k];
17: 
18:                 n--; // one gone
19:                 j--; // [1 1 2 3] -> [1 2 3]
20:                      // j--^         j--^
21:                      // j will go to 3 on the next loop
22:                      // so decrement j so it starts from 2 again
23:             }
24:         }
25:     }
26: 
27:     for (int i = 0; i < n; i++) printf("%d ", arr[i]);
28:     putchar('\n');
