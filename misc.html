Convert decimal to base

 1: #include <stdio.h>
 2: 
 3: char symbols[36]; // maximum base 36; digits (0-9) and alpha (A-Z)
 4: 
 5: void dec_to_base(int n, int base) {
 6:     if (!n) return;
 7: 
 8:     dec_to_base(n/base, base);
 9:     putchar(symbols[n % base]);    // least significant bit so printing last
10: }
11: 
12: int main(void) {
13:     int base, n;
14:     scanf("%d %d", &base, &n);
15: 
16:     if (base < 2 || base > 36) {
17:         printf("Only 2 to 36 bases are supported.\n");
18:         return 0;
19:     }
20: 
21:     if (!n) {
22:         printf("0\n");
23:         return 0;
24:     }
25: 
26:     // initializing symbols
27:     for (int i = 0; i < base; i++)
28:         symbols[i] = (i >= 10) ? 'A' + i-10 : '0' + i;
29: 
30:     dec_to_base(n, base);
31:     putchar('\n');
32: }

Armstrong number

 1: #include <stdio.h>
 2: #include <math.h>
 3: 
 4: int main() {
 5:     int n; scanf("%d", &n);
 6: 
 7:     int n_digs = log10(n) + 1;
 8: 
 9:     int sum = 0;
10:     for (int i = n; i; i /= 10)
11:         sum += pow(i % 10, n_digs);
12: 
13:     if (sum == n)
14:         printf("Yes\n");
15:     else
16:         printf("No\n");
17: }

Iterative

 1: #include <stdio.h>
 2: #include <ctype.h>
 3: 
 4: int main() {
 5:     char str[50];
 6:     int len = 0;
 7:     for (; (str[len] = getchar()) != '\n'; len++);
 8:     str[len] = 0;
 9: 
10:     // reverse string
11:     for (int i = 0; i < len/2; i++) {
12:         char tmp = str[i];
13:         str[i] = str[len-1 - i];
14:         str[len-1 - i] = tmp;
15:     }
16:     puts(str);
17: 
18:     // reverse word by word
19:     for (int i = 0, j = 0; i <= len; i++) {
20:         if ((str[i] == ' ' || str[i] == 0) && isalpha(str[i-1])) {
21:             for (int k = 0, l = i-j; k < l/2; k++) {
22:                 char tmp = str[j+k];
23:                 str[j+k] = str[i-1 - k];
24:                 str[i-1 - k] = tmp;
25:             }
26:             // skipping extra spaces
27:             for (; str[i] == ' '; i++);
28:             j = i;
29:         }
30:     }
31:     puts(str);
32: }

Recursive

 1: #include <stdio.h>
 2: 
 3: char* rev_sen(char *dup, char *s) {
 4:     if (dup[0] == '\0') return s;
 5:     if (dup[0] == ' ') dup += 1; // skipping the space
 6: 
 7:     int i;
 8:     for (i = 0; dup[i] != ' ' && dup[i] != '\0'; i++);
 9: 
10:     s = rev_sen(dup+i, s);
11: 
12:     for (i = 0; dup[i] != ' ' && dup[i] != '\0'; s[i++] = dup[i]);
13:     s[i] = ' ';
14: 
15:     return s+i+1;
16: }
17: 
18: int main() {
19:     char s[100];
20:     int len;
21:     for (len = 0; (s[len] = getchar()) != '\n'; len++);
22:     s[len] = '\0';
23: 
24:     char dup[100]; // duplicate
25:     for (int i = 0; (dup[i] = s[i]) != '\0'; i++);
26: 
27:     char *after_end = rev_sen(dup, s);
28:     *(after_end-1) = '\0';
29: 
30:     puts(s);
31: }
