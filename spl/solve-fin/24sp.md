2. WAP
   ```c
   #include <stdio.h>

   struct Patient {
      char name[50];
      int age;
      float height, weight, BMI;
   };

   int main() {
      int n = 100;
      struct Patient ar[n];
      int min = 999, indx = 0;
      for (int i = 0; i < n; i++) {
         scanf(" %[^\n]", ar[i].name);
         scanf("%d%f%f", &ar[i].age,
                           &ar[i].height,
                           &ar[i].weight);
         ar[i].BMI = ar[i].weight / ar[i].height;

         if (ar[i].age < min) {
               min = ar[i].age;
               indx = i;
         }
      }

      puts(ar[indx].name);
      printf("%d\n", ar[indx].age);
      printf("%f\n", ar[indx].height);
      printf("%f\n", ar[indx].weight);
      printf("%f\n", ar[indx].BMI);
   }
   ```
