## 03

1. Rewrite using do-while
   ```c:line-numbers
   #include <stdio.h>
   void main() {
       int n = 3, i, j, sum = 0;

       i = 0;
       do {
           j = 0;
           do {
               if (i == j) sum += i + j;
               else if (i > j) sum += i + n;
               else sum += n - j;
               j++;
           } while (j <= i);
           i++;
       } while (i < n);
   }
   ```

           }
       }

2. WAP
   ```c:line-numbers
   #include <stdio.h>

   int main() {
       int n; scanf("%d", &n);

       int arr[100];
       for (int i = 0; i < n; i++) scanf("%d", &arr[i]);

       printf("Index   Value\n");
       printf("-----   -----\n");
       for (int i = 0; i < n; i++)
           printf("%5d   %5d\n", i, arr[i]);

       int sum = 0;
       int count = 0;
       for (int i = 1; i < n; i += 2) {
           sum += arr[i];
           count++;
       }

       float avg = (float)sum / count;
       printf("avg: %f\n", avg);
   }
   ```
