## 01

```c
#include <stdio.h>

struct student {
    char name[50];
    int id;
    float cg;
};

int main() {
    int n; scanf("%d", &n);
    struct student ar[n];
    float max = 0;
    int mi = 0;
    for (int i = 0; i < n; i++) {
        scanf(" %[^\n]", ar[i].name);
        scanf("%d %f", &ar[i].id, &ar[i].cg);

        if (ar[i].cg > max) {
            max = ar[i].cg;
            mi = i;
        }
    }

    puts(ar[mi].name);
    printf("%d\n%.2f\n", ar[mi].id, ar[mi].cg);
}
```

#include <stdio.h>

int convBase(int n) {
    if (n == 0) return 0;
    return convBase(n / 6) * 10 + n % 6;
}

int sum_of_digs(int n) {
    return n == 0 ? 0 : n % 10 + sum_of_digs(n / 10);
}

int sum_of_f_n_l(int n) {
    int sum = n % 10,
        a = sum;
    for (n /= 10; n; n /= 10)
        a = n % 10;
    return sum + a;
}

int is_prime(int n) {
    if (n <= 1) return 0;
    for (int i = 2; i <= n/2; i++)
        if (n % i == 0) return 0;
    return 1;
}

int primeChecker(int *a, int *b) {
    int count = 0;
    for (; *a <= *b; *a++)
        if (is_prime(*a)) count++;
    return count;
}

int main() {
    int n; scanf("%d", &n);
    int zero = 0;
    int s = sum_of_f_n_l(n);
    if (sum_of_digs(convBase(n)) % 5 == 0)
        puts("YES");
    else if (primeChecker(&zero, &s) > 4)
        puts("YES");
    else
        puts("NO");
}
