 3: int is_prime(int n) {
 4:     if (n < 2) return 0;
 5:     for (int i = 2, lim = n; i < lim; lim /= i++)
 6:         if (n % i == 0) return 0;
 7:     return 1;
 8: }
 9: 
10: int is_pal(int n) {
11:     int rev = 0;
12:     for (int i = n; i > 0; i /= 10)
13:         rev = rev*10 + i%10;
14:     return n == rev;
15: }
16: 
17: int sum_of_digs(int n) {
18:     int sum = 0;
19:     for (; n > 0; n /= 10)
20:         sum += n%10;
21:     return sum;
22: }
23: 
24: int main() {
