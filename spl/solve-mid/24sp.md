## 03

1. Pattern '2'
   ```c:line-numbers
   #include <stdio.h>
   
   int main() {
       int n; scanf("%d", &n);
    
       for(int i = 0; i < n; i++)
           putchar('*');
       putchar('\n');
    
       for(int i = 1; i < n/2; i++) {
           for (int j = 0; j < n-1; j++) putchar(' ');
           printf("*\n");
       }
    
       for(int i = 0; i < n; i++)
           putchar('*');
       putchar('\n');
    
       for(int i = n/2+1; i < n-1; i++)
           printf("*\n");
    
       for(int i = 0; i < n; i++)
           putchar('*');
       putchar('\n');
   }
   ```
