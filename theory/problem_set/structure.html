 1: #include <stdio.h>
 2: 
 3: typedef struct {
 4:     float real, img;
 5: } complex;
 6: 
 7: complex mul(complex a, complex b) {
 8:     return (complex) {
 9:         (a.real*b.real)-(a.img*b.img),
10:         (a.real*b.img)+(a.img*b.real)
11:     };
12: }
13: 
14: complex conjugate(complex a) {
15:     return (complex) { a.real, -a.img };
16: }
17: 
18: complex div(complex a, complex b) {
19:     complex num = mul(a, conjugate(b));
20:     complex denom = mul(b, conjugate(b));
21:     return (complex) {
22:         num.real/denom.real, num.img/denom.real
23:     };
24: }
25: 
26: void print_complex(complex a) {
27:     printf("(%.2f%+.2fi)", a.real, a.img);
28: }
29: 
30: int main() {
31:     complex a, b;
32:     scanf("%f%f", &a.real, &a.img);
33:     scanf("%f%f", &b.real, &b.img);
34: 
35:     print_complex(a);
36:     putchar('/');
37:     print_complex(b);
38:     putchar('=');
39:     print_complex(div(a, b));
40:     putchar(\n);
41: }
