```c
#include <stdio.h>

int checkerFunc1(int n) {
    int d, odd = 0, evn = 0;
    for (; n; n /= 10) {
        d = n % 10;
        if (d & 1) odd += d;
        else       evn += d;
    }
    return (odd < evn) ? 1 : -1;
}

int primeChecker(int n) {
    if (n <= 1) return -1;
    for (int i = 2; i <= n / 2; i++)
        if (n % i == 0) return -1;
    return 1;
}

int DemogorgonNumber(int n) {
    return (checkerFunc1(n) > 0 && primeChecker(n) > 0) ? 1 : -1;
}

int main() {
    int a, b; scanf("%d%d", &a, &b);
    for (; a <= b; a++) {
        if (DemogorgonNumber(a) > 0)
            printf("%d ", a);
    }
    putchar('\n');
}
```

```c
#include <stdio.h>

int fib(int n) {
    if (n == 1) return 0;
    if (n == 2) return 1;
    return fib(n-1) + fib(n-2);
}

void print_fib(int a, int n) {
    if (a > n) return;
    int f = fib(a);
    printf("%d ", f);
    if (a < n) {
        if (f == 0)     putchar('&');
        else if (f & 1) putchar('#');
        else            putchar('$');
        putchar(' ');
    }
    print_fib(a+1, n);
}

int main() {
    int n; scanf("%d", &n);
    print_fib(1, n);
    putchar('\n');
}
```

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    int id;
    char name[50];
    char birth[11];
    char house[11];
    int spent[3];
    float avg;
} customer;

int substringChecker(char *str, char *sub) {
    for (char *a, *b; *str; str++) {
        for (a = str, b = sub; *a == *b && *b; a++, b++)
            ;
        if (*b == '\0') return 1;
    }
    return 0;
}

int main() {
    int n; scanf("%d", &n);
    customer ar[n];
    int si = n, gi = n;
    float smax = 0, gmax = 0;
    for (int i = 0; i < n; i++) {
        scanf(" %[^\n]", ar[i].name);
        scanf(" %[^\n]", ar[i].birth);
        scanf(" %[^\n]", ar[i].house);
        scanf("%d%d%d",
              ar[i].spent, ar[i].spent+1, ar[i].spent+2);

        ar[i].avg = (ar[i].spent[0]
                   + ar[i].spent[1]
                   + ar[i].spent[2]) / 3.0;

        if (strcmp(ar[i].house, "Slytherin") == 0) {
            if (substringChecker(ar[i].name, "rus")
                && ar[i].avg > smax)
            {
                smax = ar[i].avg;
                si = i;
            }
        } else {
            if (substringChecker(ar[i].name, "est")
                && ar[i].avg > gmax)
            {
                gmax = ar[i].avg;
                gi = i;
            }
        }
    }

    puts("Gryffindor:");
    if (gi == n)
        puts("Nil");
    else
        printf("%s | %s | %.2f\n",
               ar[gi].name, ar[gi].birth, ar[gi].avg);

    puts("Slytherin:");
    if (si == n)
        puts("Nil");
    else
        printf("%s | %s | %.2f\n",
               ar[si].name, ar[si].birth, ar[si].avg);
}
```
