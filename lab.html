 3: 
 4: #define ERROR_MARGIN 0.1
 5: 
 6: int main() {
 7:     // e.g.
 8:     // indx:  0 1   2 3    4 5
 9:     // cord: (1,1) (3,5) (-5,5)
10:     double tri_cords[3*2];
11:     for (int i = 0; i < 3; i++)
12:         scanf("%lf %lf", tri_cords + i*2, tri_cords + i*2 + 1);
13: 
14:     // sides
15:     double a = sqrt(pow(tri_cords[0] - tri_cords[2], 2)
16:                     + pow(tri_cords[1] - tri_cords[3], 2));
17:     double b = sqrt(pow(tri_cords[2] - tri_cords[4], 2)
18:                     + pow(tri_cords[3] - tri_cords[5], 2));
19:     double c = sqrt(pow(tri_cords[0] - tri_cords[4], 2)
20:                     + pow(tri_cords[1] - tri_cords[5], 2));
21: 
22:     // checking invalid
23:     if ((a + b <= c) || (b + c <= a) || (c + a <= b)) {
24:         printf("Invalid\n");
25:         return 0;
26:     }
27: 
28:     if (fabs(a - b) <= ERROR_MARGIN && fabs(b - c) <= ERROR_MARGIN
29:         && fabs(c - a) <= ERROR_MARGIN)
30:         printf("Equilateral\n");
31:     else if (fabs(a - b) <= ERROR_MARGIN
32:              || fabs(b - c) <= ERROR_MARGIN
33:              || fabs(c - a) <= ERROR_MARGIN)
34:         printf("Isosceles\n");
35:     else
36:         printf("Scalene\n");
37: }
