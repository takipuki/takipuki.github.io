    student ar[50];
    for (int i = 0; i < 50; i++) {
        scanf(" %[^\n]", ar[i].name);
        scanf("%d", &ar[i].id);
        float total = 0;
        for (int j = 0; j < 5; j++) {
            scanf("%f", &ar[i].marks[j]);
            total += ar[i].marks[j];
        }
    }

    for (int j = 0; j < 5; j++) {
        float max = 0;
        int k = 0;
        for (int i = 0; i < 50; i++)
            if (ar[i].marks[j] > max) {
                max = ar[i].marks[j];   
                k = i;
            }

}
